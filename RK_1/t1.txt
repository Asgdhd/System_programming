; Работаем в папке RK_1
cd RK_1

; Создание каталога sandbox
mkdir sandbox

; Переход в каталог sandbox
cd sandbox

; Создание файла @@@ со списком процессов от пользователя root
ps uax | grep root > '@@@'

; Создание файла !!! с именами файлов из /bin, начинающихся на "b"
ls /bin/b* > '!!!'

; Создание файла """ с версией ядра ОС
uname -a > '"""'

; Добавление содержимого файла @@@ в конец файла """
cat '@@@' >> '"""'

; Создание жесткой ссылки на файл !!!
ln '!!!' 'hardlink!!!'

; Создание архива с тремя файлами
tar -cf archive.tar '@@@' '!!!' '"""'

; Создание каталога subdir
mkdir subdir

; Перемещение архива в subdir с переименованием в текущую дату
mv archive.tar subdir/$(date +%Y-%m-%d)
